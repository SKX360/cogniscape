% ---------- Volume command (place in preamble) ----------
\usepackage{etoolbox} % small dependency; safe and useful
\newcounter{volume}
\renewcommand\thevolume{\Roman{volume}} % change to \arabic if you prefer

\newcommand{\volume}[1]{%
  \cleardoublepage
  \refstepcounter{volume}% increment counter and make it referencable
  \thispagestyle{plain}%
  \begingroup
    \centering
    \vspace*{4cm}%
    {\Huge\bfseries Volume~\thevolume\par}%
    \vskip2em
    {\LARGE #1\par}%
    \vskip4em
  \endgroup
  % Add to TOC at part level (simple, robust form)
  \addcontentsline{toc}{part}{Volume~\thevolume: #1}%
  \cleardoublepage
}
